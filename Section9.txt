void setup(){
  noLoop();
}

void draw(){
  int[] stack = new int[5];
  println("Popped:");
  for (int i=0;i<7;i++){
    if (i < 5){
      stack[i] = i + 1;
    }else{
      print(stack[stack.length - 1], " ");
      stack[stack.length - 1] = i + 1;
    }
  }
  println("stack:");
  for (int i=0;i<stack.length;i++){
    print(stack[i], " ");
  }
  println();
}
